```python
# 手写一个缓存装饰器，支持按参数缓存、设置过期时间
import time
from functools import wraps
from typing import Any, Callable

def cached(ttl: int = 60):
    """带过期时间的缓存装饰器"""
    def decorator(func: Callable) -> Callable:
        cache = {}
        @wraps(func)
        def wrapper(*args, **kwargs):
            key = (args, tuple(sorted(kwargs.items())))
            now = time.time()
            if key in cache:
                result, ts = cache[key]
                if now - ts < ttl:
                    return result
            result = func(*args, **kwargs)
            cache[key] = (result, now)
            return result
        return wrapper
    return decorator
```

### 1.2 多任务并发（重中之重）

#### 三种并发模型对比

| 模型 | 适用场景 | 核心模块 | 性能特点 |
|------|---------|---------|---------|
| 多线程 `threading` | I/O 密集 + 需共享内存 | `threading`, `concurrent.futures` | 受 GIL 限制，CPU 密集时退化为串行 |
| 多进程 `multiprocessing` | CPU 密集型计算 | `multiprocessing`, `ProcessPoolExecutor` | 绕过 GIL，内存开销大 |
| 协程 `asyncio` | 高并发 I/O（网络请求、数据库） | `asyncio`, `aiohttp`, `aiomysql` | 单线程，数万并发连接 |

#### asyncio 深入（面试高频）

**核心概念**：
- **事件循环 (Event Loop)**：调度协程执行的核心引擎
- **协程 (Coroutine)**：用 `async def` 定义的函数，可被 `await` 挂起
- **Task**：将协程包装为可调度的任务
- **Future**：表示异步操作的最终结果

**必须掌握的 API**：
```python
import asyncio

# 1. 并发执行多个协程
async def main():
    # gather: 并发执行，全部完成返回结果列表
    results = await asyncio.gather(task1(), task2(), task3())

    # as_completed: 哪个先完成就先处理哪个
    for coro in asyncio.as_completed([task1(), task2()]):
        result = await coro
        print(result)

# 2. 超时控制
async def with_timeout():
    try:
        result = await asyncio.wait_for(slow_task(), timeout=5.0)
    except asyncio.TimeoutError:
        print("超时了")

# 3. 信号量控制并发数
async def limited_concurrency():
    sem = asyncio.Semaphore(10)  # 最多10个并发
    async def worker(url):
        async with sem:
            return await fetch(url)
    await asyncio.gather(*[worker(url) for url in urls])
```

**常见面试题**：
1. `asyncio.gather` vs `asyncio.wait` 的区别？
   - `gather` 返回结果列表，任一异常默认取消其他任务
   - `wait` 返回 (done, pending) 集合，可配置异常策略
2. 协程中调用同步阻塞代码怎么办？
   - `loop.run_in_executor(executor, func)` 放到线程/进程池
3. 如何理解 `await` 的"挂起"？
   - 遇到 I/O 时让出控制权，事件循环执行其他任务，I/O 完成后恢复

```python
import asyncio
import aiohttp
import time

async def download(session, url, sem):
    async with sem:
        async with session.get(url) as resp:
            return await resp.text()

async def batch_download(urls, concurrency=20):
    sem = asyncio.Semaphore(concurrency)
    async with aiohttp.ClientSession() as session:
        tasks = [download(session, url, sem) for url in urls]
        return await asyncio.gather(*tasks)

# 测试
start = time.perf_counter()
asyncio.run(batch_download(["https://httpbin.org/delay/1"] * 100, 20))
print(f"耗时: {time.perf_counter() - start:.2f}s")  # 约5秒（100/20*1s）
```

#### 进阶：连接管理器（多人聊天室）

```python
import asyncio
from typing import Dict, Set
from fastapi import WebSocket

class ConnectionManager:
    """管理所有 WebSocket 连接"""

    def __init__(self):
        # room_id -> set of websocket connections
        self.rooms: Dict[str, Set[WebSocket]] = {}
        self._lock = asyncio.Lock()

    async def connect(self, room_id: str, websocket: WebSocket):
        await websocket.accept()
        async with self._lock:
            if room_id not in self.rooms:
                self.rooms[room_id] = set()
            self.rooms[room_id].add(websocket)

    async def disconnect(self, room_id: str, websocket: WebSocket):
        async with self._lock:
            self.rooms.get(room_id, set()).discard(websocket)
            if not self.rooms.get(room_id):
                self.rooms.pop(room_id, None)

    async def broadcast(self, room_id: str, message: dict):
        """广播消息到房间内所有连接"""
        room = self.rooms.get(room_id, set()).copy()
        dead = []
        for ws in room:
            try:
                await ws.send_json(message)
            except Exception:
                dead.append(ws)
        # 清理断开的连接
        for ws in dead:
            await self.disconnect(room_id, ws)

    async def send_to(self, room_id: str, websocket: WebSocket, message: dict):
        """发送给指定连接"""
        try:
            await websocket.send_json(message)
        except Exception:
            await self.disconnect(room_id, websocket)

manager = ConnectionManager()

@router.websocket("/ws/room/{room_id}/{user_id}")
async def room_chat(websocket: WebSocket, room_id: str, user_id: str):
    await manager.connect(room_id, websocket)
    await manager.broadcast(room_id, {"type": "system", "content": f"{user_id} 加入了房间"})

    try:
        while True:
            data = await websocket.receive_json()
            message = {
                "type": "message",
                "user_id": user_id,
                "content": data.get("content", ""),
                "timestamp": datetime.now().isoformat(),
            }
            # 持久化到数据库
            await save_message(room_id, message)
            # 广播给房间所有人
            await manager.broadcast(room_id, message)
    except WebSocketDisconnect:
        await manager.disconnect(room_id, websocket)
        await manager.broadcast(room_id, {"type": "system", "content": f"{user_id} 离开了房间"})
```

### 3.3 生产级 WebSocket 需解决的问题

| 问题 | 解决方案 |
|------|---------|
| **断线重连** | 客户端实现指数退避重连；服务端支持 session 恢复 |
| **心跳保活** | 定期 ping/pong，超时断开（FastAPI: `websocket.receive_text()` 超时检测） |
| **消息确认** | 消息带唯一 ID，客户端 ACK，未 ACK 的消息重发 |
| **消息顺序** | 消息带序列号，客户端排序 |
| **连接数过多** | 使用 Redis Pub/Sub 跨进程广播（多 worker 场景） |
| **鉴权** | URL 参数传 token 或首条消息认证（WebSocket 不支持自定义 Header） |

#### 心跳机制实现

```python
import asyncio
from fastapi import WebSocket

HEARTBEAT_INTERVAL = 30  # 秒
HEARTBEAT_TIMEOUT = 60   # 秒

async def heartbeat_loop(websocket: WebSocket):
    """心跳检测协程"""
    try:
        while True:
            await websocket.send_json({"type": "ping"})
            await asyncio.sleep(HEARTBEAT_INTERVAL)
    except Exception:
        pass

@router.websocket("/ws/heartbeat")
async def ws_with_heartbeat(websocket: WebSocket):
    await websocket.accept()
    heartbeat_task = asyncio.create_task(heartbeat_loop(websocket))
    try:
        while True:
            try:
                data = await asyncio.wait_for(
                    websocket.receive_json(),
                    timeout=HEARTBEAT_TIMEOUT
                )
                if data.get("type") == "pong":
                    continue  # 心跳响应，跳过
                # 处理业务消息...
            except asyncio.TimeoutError:
                logger.warning("Client heartbeat timeout, closing")
                break
    finally:
        heartbeat_task.cancel()
        await websocket.close()
```

### 3.4 跨 Worker 广播（Redis Pub/Sub）

```python
import redis.asyncio as redis
import json

class RedisPubSubManager:
    """多进程/多实例场景下的 WebSocket 消息广播"""

    def __init__(self, redis_url: str):
        self.redis = redis.from_url(redis_url)
        self.pubsub = self.redis.pubsub()
        self.local_connections: Dict[str, Set[WebSocket]] = {}

    async def subscribe(self, channel: str):
        await self.pubsub.subscribe(channel)
        async for message in self.pubsub.listen():
            if message["type"] == "message":
                data = json.loads(message["data"])
                await self._broadcast_local(channel, data)

    async def publish(self, channel: str, message: dict):
        """任何 worker 调用 publish，所有 worker 都能收到"""
        await self.redis.publish(channel, json.dumps(message))

    async def _broadcast_local(self, channel: str, message: dict):
        for ws in self.local_connections.get(channel, set()).copy():
            try:
                await ws.send_json(message)
            except Exception:
                self.local_connections[channel].discard(ws)
```

### 6.2 MySQL 深入（面试重点）

#### 索引原理（B+ 树）

```
B+ 树索引结构：
         [非叶子节点: 10, 20, 30]           ← 只做路由，不存数据
        /         |         \
  [5,8,10]   [12,15,20]   [22,25,30]        ← 叶子节点，存数据行指针
    ↕            ↕             ↕
  双向链表连接所有叶子节点                    ← 支持范围查询

聚簇索引（InnoDB）：
  叶子节点直接存储完整数据行
  主键就是聚簇索引，一张表只能有一个

二级索引（辅助索引）：
  叶子节点存储主键值
  查询非索引列需要"回表"（二级索引 → 主键 → 聚簇索引取数据）

覆盖索引：
  查询的列全部在索引中，无需回表
  EXPLAIN 中 Extra 显示 "Using index"
```

#### EXPLAIN 执行计划（面试必问）

```sql
EXPLAIN SELECT u.name, o.total
FROM users u
JOIN orders o ON u.id = o.user_id
WHERE u.status = 'active' AND o.created_at > '2024-01-01';
```

**关键字段解读**：

| 字段 | 关注点 |
|------|--------|
| `type` | 连接类型，从好到差：`system > const > eq_ref > ref > range > index > ALL` |
| `key` | 实际使用的索引，`NULL` 表示全表扫描 |
| `rows` | 预估扫描行数，越小越好 |
| `Extra` | `Using index`(覆盖索引)✅ / `Using filesort`(文件排序)⚠️ / `Using temporary`(临时表)❌ |
| `possible_keys` | 可能使用的索引 |

#### 事务与隔离级别

```
四种隔离级别（从低到高）：

READ UNCOMMITTED  → 脏读 ✅  不可重复读 ✅  幻读 ✅
READ COMMITTED    → 脏读 ❌  不可重复读 ✅  幻读 ✅    (Oracle默认)
REPEATABLE READ   → 脏读 ❌  不可重复读 ❌  幻读 ✅    (MySQL InnoDB默认)
SERIALIZABLE      → 脏读 ❌  不可重复读 ❌  幻读 ❌

InnoDB 的 REPEATABLE READ 通过 MVCC + Next-Key Lock 解决了大部分幻读问题
```

**MVCC（多版本并发控制）原理**：
```
每行数据有隐藏字段：
  - trx_id：最近修改的事务ID
  - roll_pointer：指向 undo log 中旧版本的指针

Read View（快照）：
  事务在第一次 SELECT 时创建 Read View
  包含：创建时活跃事务列表、最小活跃事务ID、下一个事务ID

可见性判断：
  数据版本 trx_id < min_trx_id  → 可见（已提交）
  数据版本 trx_id >= max_trx_id → 不可见（未开始）
  数据版本在活跃列表中           → 不可见（未提交）
```

#### 慢查询优化实战

```sql
-- ❌ 慢查询：分页性能差（大表 OFFSET 扫描大量行）
SELECT * FROM orders ORDER BY created_at DESC LIMIT 10 OFFSET 100000;
-- 实际扫描 100010 行，丢弃前 100000 行

-- ✅ 优化1：游标分页（基于上一页最后一条的 ID）
SELECT * FROM orders WHERE id < 100000 ORDER BY id DESC LIMIT 10;

-- ❌ 慢查询：LIKE 前缀模糊
SELECT * FROM users WHERE name LIKE '%张三%';

-- ✅ 优化2：全文索引 或 搜索引擎（Elasticsearch）
ALTER TABLE users ADD FULLTEXT INDEX ft_name(name);
SELECT * FROM users WHERE MATCH(name) AGAINST('张三' IN BOOLEAN MODE);

-- ❌ 慢查询：N+1 问题
SELECT * FROM users;           -- 1次查询
SELECT * FROM orders WHERE user_id = ?;  -- N次查询

-- ✅ 优化3：JOIN 或子查询一次性获取
SELECT u.*, o.* FROM users u LEFT JOIN orders o ON u.id = o.user_id;

-- ✅ 优化4：添加复合索引
ALTER TABLE orders ADD INDEX idx_user_created (user_id, created_at);
```

#### MySQL 高可用架构

```
主从复制（Master-Slave）：
  Master ──binlog──→ Slave1
                  └─→ Slave2
  用途：读写分离、备份、灾备

读写分离：
  写操作 → Master
  读操作 → Slave（通过代理中间件：MyCat、ProxySQL）

分库分表（数据量过大时）：
  垂直拆分：按业务拆到不同数据库（user_db, order_db）
  水平拆分：同一张表按规则分到多个库/表（hash(id) % 16）
  工具：ShardingSphere、Vitess
```

#### 何时使用 SQLite

```
✅ 适合：
  - 开发和测试环境（零配置，快速启动）
```

#### Python 中使用 SQLite

```python
import sqlite3
from contextlib import contextmanager

# 上下文管理器封装连接
@contextmanager
def get_db(db_path: str = "app.db"):
    conn = sqlite3.connect(db_path)
    conn.row_factory = sqlite3.Row  # 结果可通过列名访问
    conn.execute("PRAGMA journal_mode=WAL")  # WAL 模式：读写并发
    conn.execute("PRAGMA foreign_keys=ON")
    try:
        yield conn
        conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:
        conn.close()

# 使用
with get_db() as conn:
    conn.execute("""
        CREATE TABLE IF NOT EXISTS users (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            email TEXT UNIQUE NOT NULL,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    """)
    conn.execute(
        "INSERT INTO users (name, email) VALUES (?, ?)",
        ("张三", "zhangsan@example.com")
    )
    users = conn.execute("SELECT * FROM users").fetchall()
    for u in users:
        print(u["name"], u["email"])
```

#### SQLite 性能优化

| 优化项 | 命令 | 效果 |
|--------|------|------|
| WAL 模式 | `PRAGMA journal_mode=WAL` | 读写并发，不互相阻塞 |
| 内存映射 | `PRAGMA mmap_size=268435456` | 将数据库映射到内存，加速读取 |
| 同步模式 | `PRAGMA synchronous=NORMAL` | 减少 fsync，提升写入速度 |
| 缓存大小 | `PRAGMA cache_size=-64000` | 64MB 缓存 |
| 批量插入 | 使用 `executemany` + 事务 | 比逐条插入快 100x |

#### 数据清洗流水线

```python
import re
from typing import List
from dataclasses import dataclass

@dataclass
class Document:
    content: str
    metadata: dict  # source, page, title, etc.

class DocumentCleaner:
    """文档清洗器"""

    def clean(self, doc: Document) -> Document:
        text = doc.content
        # 1. 去除多余空白（保留单个空格和换行）
        text = re.sub(r'[ \t]+', ' ', text)
        text = re.sub(r'\n{3,}', '\n\n', text)

        # 2. 去除页眉页脚（PDF 常见）
        text = re.sub(r'^.*?第\d+页.*?$', '', text, flags=re.MULTILINE)

        # 3. 去除特殊字符和乱码
        text = re.sub(r'[\x00-\x08\x0b\x0c\x0e-\x1f]', '', text)

        # 4. 修复常见 OCR 错误
        text = text.replace('。一', '。')
        text = text.replace('，,', '，')

        # 5. 去除过短的无意义段落
        paragraphs = [p.strip() for p in text.split('\n\n') if len(p.strip()) > 10]

        doc.content = '\n\n'.join(paragraphs)
        return doc
```

```python
    """评估切片质量"""
    return {
        "total_chunks": len(chunks),
        "avg_length": sum(len(c) for c in chunks) / len(chunks),
        "min_length": min(len(c) for c in chunks),
        "max_length": max(len(c) for c in chunks),
        "empty_chunks": sum(1 for c in chunks if not c.strip()),
        "too_short": sum(1 for c in chunks if len(c) < 50),  # 太短可能无意义
        "too_long": sum(1 for c in chunks if len(c) > 2000), # 太长可能不精确
    }
```

```python
from abc import ABC, abstractmethod
from typing import List
import numpy as np

class BaseEmbedding(ABC):
    @abstractmethod
    async def embed_documents(self, texts: List[str]) -> List[List[float]]:
        pass

    @abstractmethod
    async def embed_query(self, text: str) -> List[float]:
        pass

class OpenAIEmbedding(BaseEmbedding):
    def __init__(self, model: str = "text-embedding-3-small", api_key: str = None):
        from openai import AsyncOpenAI
        self.client = AsyncOpenAI(api_key=api_key)
        self.model = model

    async def embed_documents(self, texts: List[str]) -> List[List[float]]:
        # 批量处理，每次最多 100 条
        all_embeddings = []
        for i in range(0, len(texts), 100):
            batch = texts[i:i+100]
            resp = await self.client.embeddings.create(input=batch, model=self.model)
            all_embeddings.extend([e.embedding for e in resp.data])
        return all_embeddings

    async def embed_query(self, text: str) -> List[float]:
        resp = await self.client.embeddings.create(input=[text], model=self.model)
        return resp.data[0].embedding

class BGEEmbedding(BaseEmbedding):
    """本地部署 BGE 模型"""
    def __init__(self, model_name: str = "BAAI/bge-large-zh-v1.5"):
        from sentence_transformers import SentenceTransformer
        self.model = SentenceTransformer(model_name)

    async def embed_documents(self, texts: List[str]) -> List[List[float]]:
        embeddings = self.model.encode(texts, normalize_embeddings=True)
        return embeddings.tolist()

    async def embed_query(self, text: str) -> List[float]:
        # BGE 查询时加 "为这个句子生成表示以用于检索：" 前缀
        text = f"为这个句子生成表示以用于检索：{text}"
        embedding = self.model.encode([text], normalize_embeddings=True)
        return embedding[0].tolist()
```

### 7.6 RAG 检索优化（进阶）

#### 混合检索（向量 + 关键词）

```python
class HybridRetriever:
    """向量检索 + BM25 关键词检索的混合方案"""

    def __init__(self, vector_store, documents: List[Document]):
        self.vector_store = vector_store
        # 构建 BM25 索引
        from rank_bm25 import BM25Okapi
        tokenized = [doc.content.split() for doc in documents]
        self.bm25 = BM25Okapi(tokenized)
        self.documents = documents

    async def search(self, query: str, top_k: int = 5, alpha: float = 0.7) -> List[Document]:
        """
        alpha: 向量检索权重 (1-alpha 为 BM25 权重)
        """
        # 向量检索
        vector_results = await self.vector_store.search(query, top_k=top_k * 2)

        # BM25 检索
        bm25_scores = self.bm25.get_scores(query.split())
        bm25_top = sorted(enumerate(bm25_scores), key=lambda x: x[1], reverse=True)[:top_k * 2]

        # RRF (Reciprocal Rank Fusion) 融合排序
        scores = {}
        for rank, (doc, score) in enumerate(vector_results):
            scores[doc.id] = scores.get(doc.id, 0) + alpha * (1 / (rank + 1))
        for rank, (idx, score) in enumerate(bm25_top):
            doc_id = self.documents[idx].id
            scores[doc_id] = scores.get(doc_id, 0) + (1 - alpha) * (1 / (rank + 1))

        ranked_ids = sorted(scores, key=scores.get, reverse=True)[:top_k]
        return [doc for doc in self.documents if doc.id in ranked_ids]
```

```python
class Reranker:
    """用 Cross-Encoder 模型对检索结果重排序"""

    def __init__(self, model_name: str = "BAAI/bge-reranker-v2-m3"):
        from sentence_transformers import CrossEncoder
        self.model = CrossEncoder(model_name)

    def rerank(self, query: str, documents: List[str], top_k: int = 3) -> List[tuple]:
        """返回 (分数, 文档) 的排序列表"""
        pairs = [(query, doc) for doc in documents]
        scores = self.model.predict(pairs)
        scored = sorted(zip(scores, documents), key=lambda x: x[0], reverse=True)
        return scored[:top_k]
```

#### 手写 ReAct Agent（理解原理）

```python
import json
from typing import Any, Callable, Dict, List
from dataclasses import dataclass, field

@dataclass
class Tool:
    name: str
    description: str
    parameters: dict
    func: Callable

@dataclass
class Message:
    role: str  # "system" | "user" | "assistant" | "tool"
    content: str
    tool_calls: List[dict] = field(default_factory=list)
    tool_call_id: str = None

class SimpleReActAgent:
    def __init__(self, llm_client, tools: List[Tool], max_iterations: int = 10):
        self.llm = llm_client
        self.tools = {t.name: t for t in tools}
        self.max_iterations = max_iterations

    def _build_system_prompt(self) -> str:
        tool_descriptions = "\n".join([
            f"- {t.name}: {t.description} | 参数: {json.dumps(t.parameters, ensure_ascii=False)}"
            for t in self.tools.values()
        ])
        return f"""你是一个智能助手，可以使用以下工具来帮助用户：
{tool_descriptions}

回复格式：
当你需要使用工具时，输出 JSON：
{{"tool": "工具名", "parameters": {{参数}}}}

当你有最终答案时，输出：
{{"answer": "你的回答"}}
"""

    async def run(self, user_input: str) -> str:
        messages = [
            Message(role="system", content=self._build_system_prompt()),
            Message(role="user", content=user_input),
        ]

        for i in range(self.max_iterations):
            response = await self.llm.chat(messages)

            try:
                parsed = json.loads(response)
            except json.JSONDecodeError:
                return response  # LLM 直接返回了文本

            # 最终答案
            if "answer" in parsed:
                return parsed["answer"]

            # 工具调用
            tool_name = parsed.get("tool")
            params = parsed.get("parameters", {})

            if tool_name not in self.tools:
                observation = f"错误：工具 {tool_name} 不存在"
            else:
                try:
                    result = await self.tools[tool_name].func(**params)
                    observation = str(result)
                except Exception as e:
                    observation = f"工具执行错误：{str(e)}"

            messages.append(Message(role="assistant", content=response))
            messages.append(Message(role="tool", content=observation))

        return "达到最大迭代次数，无法给出完整答案。"
```

### 8.5 LangGraph Agent（生产级推荐）

```python
from langgraph.graph import StateGraph, END
from langgraph.prebuilt import ToolNode
from typing import TypedDict, Annotated, List
from langchain_core.messages import BaseMessage, HumanMessage, AIMessage
import operator

# 定义状态
class AgentState(TypedDict):
    messages: Annotated[List[BaseMessage], operator.add]

# 定义节点
async def agent_node(state: AgentState) -> AgentState:
    """Agent 推理节点"""
    response = await llm.ainvoke(state["messages"])
    return {"messages": [response]}

def should_continue(state: AgentState) -> str:
    """路由决策：继续调用工具 or 返回结果"""
    last_message = state["messages"][-1]
    if hasattr(last_message, "tool_calls") and last_message.tool_calls:
        return "tools"
    return "end"

# 构建状态图
graph = StateGraph(AgentState)

graph.add_node("agent", agent_node)
graph.add_node("tools", ToolNode(tools))

graph.set_entry_point("agent")
graph.add_conditional_edges("agent", should_continue, {
    "tools": "tools",
    "end": END,
})
graph.add_edge("tools", "agent")  # 工具执行后回到 agent

app = graph.compile()

# 运行
result = await app.ainvoke({
    "messages": [HumanMessage(content="帮我查一下公司的请假制度，如果查不到就创建一个工单")]
})
```

### 8.6 记忆系统（Memory）

```python
from typing import List
from dataclasses import dataclass

@dataclass
class MemoryItem:
    content: str
    role: str
    timestamp: float
    importance: float = 0.5

class AgentMemory:
    """三层记忆架构"""

    def __init__(self, max_short_term: int = 20):
        self.short_term: List[MemoryItem] = []   # 短期：当前对话
        self.working: dict = {}                    # 工作记忆：当前任务上下文
        self.long_term_store = None                # 长期：向量数据库

        self.max_short_term = max_short_term

    def add_message(self, content: str, role: str):
        self.short_term.append(MemoryItem(
            content=content, role=role,
            timestamp=time.time()
        ))
        # 超出短期记忆容量时，摘要压缩
        if len(self.short_term) > self.max_short_term:
            self._compress_old_messages()

    def _compress_old_messages(self):
        """将旧消息摘要压缩"""
        old = self.short_term[:len(self.short_term) - 10]
        summary = summarize([m.content for m in old])  # LLM 摘要
        self.short_term = [
            MemoryItem(content=f"[历史摘要] {summary}", role="system", timestamp=time.time())
        ] + self.short_term[-10:]

    def set_working_context(self, key: str, value: Any):
        """设置工作记忆（当前任务的临时上下文）"""
        self.working[key] = value

    async def recall_from_long_term(self, query: str, top_k: int = 3) -> List[str]:
        """从长期记忆中检索相关信息"""
        if self.long_term_store:
            results = await self.long_term_store.search(query, top_k=top_k)
            return [r.content for r in results]
        return []

    def get_context(self) -> str:
        """组装当前记忆上下文"""
        parts = []
        if self.working:
            parts.append(f"[当前任务上下文]\n{json.dumps(self.working, ensure_ascii=False)}")
        parts.append("[对话历史]")
        for m in self.short_term:
            parts.append(f"  {m.role}: {m.content}")
        return "\n".join(parts)
```

#### 编写 MCP Server

```python
from mcp.server import Server
from mcp.server.stdio import stdio_server
import mcp.types as types

server = Server("my-knowledge-server")

@server.list_tools()
async def list_tools() -> list[types.Tool]:
    return [
        types.Tool(
            name="search_docs",
            description="搜索公司知识库文档",
            inputSchema={
                "type": "object",
                "properties": {
                    "query": {"type": "string", "description": "搜索关键词"},
                    "top_k": {"type": "integer", "default": 5},
                },
                "required": ["query"],
            },
        ),
        types.Tool(
            name="get_document",
            description="根据文档 ID 获取完整文档内容",
            inputSchema={
                "type": "object",
                "properties": {
                    "doc_id": {"type": "string", "description": "文档ID"},
                },
                "required": ["doc_id"],
            },
        ),
    ]

@server.call_tool()
async def call_tool(name: str, arguments: dict) -> list[types.TextContent]:
    if name == "search_docs":
        results = await rag_search(arguments["query"], arguments.get("top_k", 5))
        return [types.TextContent(type="text", text=json.dumps(results, ensure_ascii=False))]
    elif name == "get_document":
        doc = await get_doc_by_id(arguments["doc_id"])
        return [types.TextContent(type="text", text=doc.content)]
    raise ValueError(f"Unknown tool: {name}")

async def main():
    async with stdio_server() as (read, write):
        await server.run(read, write)
```

### 8.8 多 Agent 协作模式

```
常见协作模式：

1. 顺序链 (Sequential)
   Agent A → Agent B → Agent C
   场景：先研究 → 再写作 → 最后审核

2. 监督者 (Supervisor)
   Supervisor Agent 分配任务给 Worker Agents
   场景：项目经理分配子任务给不同专家

3. 辩论式 (Debate)
   Agent A ←→ Agent B 互相辩论
   场景：代码 Review、方案评审

4. 层级式 (Hierarchical)
   Manager → [Team Lead A → Workers, Team Lead B → Workers]
   场景：大型复杂任务分解
```

```python
from crewai import Agent, Task, Crew, Process

# 定义 Agent
researcher = Agent(

    goal="将研究结果整理为清晰的技术文档",
    backstory="你是一位技术写作专家，擅长将复杂技术概念用简洁语言表达",
    llm="gpt-4o",
)

reviewer = Agent(
    role="技术审核员",
    goal="审核文档的技术准确性和完整性",
    backstory="你是一位严格的技术审核员，确保文档质量达到发布标准",
    llm="gpt-4o",
)

# 定义任务
research_task = Task(
    description="调研 RAG 系统的最新进展，重点关注检索优化和评估方法",
    expected_output="一份包含 3-5 个关键发现的研究报告",
    agent=researcher,
)

write_task = Task(
    description="基于研究报告撰写技术博客文章",
    expected_output="一篇 2000 字的技术博客",
    agent=writer,
)

    description="审核博客文章的技术准确性",
    expected_output="审核意见和修改建议",
    agent=reviewer,
)

# 组建团队
crew = Crew(
    agents=[researcher, writer, reviewer],
    tasks=[research_task, write_task, review_task],
    process=Process.sequential,  # 顺序执行
    verbose=True,
)

result = crew.kickoff()
```
