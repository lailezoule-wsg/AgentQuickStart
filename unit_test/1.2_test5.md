**题目 1（事件系统）：**

```python
class EventEmitter:
    def __init__(self):
        self._listeners = {}
        self._once = set()

    def on(self, event, callback):
        if event not in self._listeners:
            self._listeners[event] = []
        self._listeners[event].append(callback)

    def off(self, event, callback):
        if event in self._listeners:
            self._listeners[event].remove(callback)

    def emit(self, event, *args, **kwargs):
        if event not in self._listeners:
            return
        for callback in self._listeners[event][:]:
            callback(*args, **kwargs)
            if id(callback) in self._once:
                self._listeners[event].remove(callback)
                self._once.discard(id(callback))

    def once(self, event, callback):
        self.on(event, callback)
        self._once.add(id(callback))

    def listener_count(self, event):
        return len(self._listeners.get(event, []))
```

```python
        self._history.append(command)

        command = self._history.pop()
```

**题目 3（日志分析器）：**

```python
import json
import re
from pathlib import Path
from datetime import datetime

class LogAnalyzer:
    LOG_PATTERN = re.compile(
        r"\[(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\] (\w+): (.+)"
    )

    def __init__(self, log_dir):
        self.log_dir = Path(log_dir)

    def parse_line(self, line):
        match = self.LOG_PATTERN.match(line.strip())
        if not match:
            return None
        timestamp_str, level, message = match.groups()
        return {
            "timestamp": datetime.strptime(timestamp_str, "%Y-%m-%d %H:%M:%S"),
            "level": level,
            "message": message,
        }

    def _all_entries(self):
        entries = []
        for log_file in self.log_dir.glob("*.log"):
            for line in log_file.read_text(encoding="utf-8").splitlines():
                entry = self.parse_line(line)
                if entry:
                    entries.append(entry)
        return entries

    def count_by_level(self):
        counts = {}
        for entry in self._all_entries():
            level = entry["level"]
            counts[level] = counts.get(level, 0) + 1
        return counts

    def errors_in_range(self, start, end):
        return [
            entry for entry in self._all_entries()
            if entry["level"] == "ERROR" and start <= entry["timestamp"] <= end
        ]

    def export_summary(self, output_path):
        summary = self.count_by_level()
        Path(output_path).write_text(
            json.dumps(summary, indent=2, ensure_ascii=False),
            encoding="utf-8"
        )
```

**题目 4（异步限流器）：**

```python
import asyncio
import time

class RateLimiter:
    def __init__(self, rate, capacity):
        self.rate = rate
        self.capacity = capacity
        self._tokens = capacity
        self._last_refill = time.monotonic()
        self._lock = asyncio.Lock()

    def _refill(self):
        now = time.monotonic()
        elapsed = now - self._last_refill
        new_tokens = elapsed * self.rate
        self._tokens = min(self.capacity, self._tokens + new_tokens)
        self._last_refill = now

    async def acquire(self):
        while True:
            async with self._lock:
                self._refill()
                if self._tokens >= 1:
                    self._tokens -= 1
                    return
            await asyncio.sleep(1.0 / self.rate)

    def available(self):
        self._refill()
        return self._tokens

async def demo():
    limiter = RateLimiter(rate=2, capacity=3)
    start = time.monotonic()
    for i in range(5):
        await limiter.acquire()
        elapsed = time.monotonic() - start
        print(f"请求 {i+1} 通过 @ {elapsed:.1f}s")

asyncio.run(demo())
```

**题目 5（Pydantic 配置验证系统）：**

```python
import json
from pathlib import Path
from pydantic import BaseModel, field_validator
from typing import Literal

class DatabaseConfig(BaseModel):
    host: str
    port: int = 5432
    name: str
    pool_size: int = 5

    @field_validator("port")
    @classmethod
    def check_port(cls, v):
        if not 1 <= v <= 65535:
            raise ValueError("port 必须在 1~65535 之间")
        return v

    @field_validator("pool_size")
    @classmethod
    def check_pool_size(cls, v):
        if not 1 <= v <= 100:
            raise ValueError("pool_size 必须在 1~100 之间")
        return v

class CacheConfig(BaseModel):
    backend: Literal["redis", "memcached"]
    ttl: int = 3600
    max_memory: str = "256MB"

    @field_validator("ttl")
    @classmethod
    def check_ttl(cls, v):
        if v <= 0:
            raise ValueError("ttl 必须大于 0")
        return v

    @field_validator("max_memory", mode="before")
    @classmethod
    def parse_memory(cls, v):
        if isinstance(v, str):
            v = v.strip().upper()
            if v.endswith("GB"):
                return str(int(float(v[:-2]) * 1024 * 1024 * 1024))
            elif v.endswith("MB"):
                return str(int(float(v[:-2]) * 1024 * 1024))
            elif v.endswith("KB"):
                return str(int(float(v[:-2]) * 1024))
        return v

class AppConfig(BaseModel):
    app_name: str
    version: str
    debug: bool = False
    database: DatabaseConfig
    cache: CacheConfig

    @classmethod
    def from_json_file(cls, path):
        data = json.loads(Path(path).read_text(encoding="utf-8"))
        return cls(**data)

    def to_json_file(self, path):
        Path(path).write_text(
            self.model_dump_json(indent=2),
            encoding="utf-8"
        )
```
