> - 第 1 题：`obj.value = 20` 通过 setter 设置成功，`print(obj.value)` 输出 20。`obj.value = -5` 触发 setter 中的检查，抛出 `ValueError`。
> - 第 3 题：`__iter__` 返回 `reversed(self._items)`，栈从底到顶为 `[1, 2, 3]`，反转后为 `[3, 2, 1]`。
> - 第 4 题：`Child.__init__` 先调用 `super().__init__()` 设 `self.name = "Base"`，再覆盖为 `self.name = "Child"`。`info()` 方法在 `Base` 中定义，但 `self` 是 `Child` 实例，`self.name` 为 `"Child"`。

> - 第 15 题：`type(self).__name__` 返回实际实例的类名，即 `"MonitoredService"`。

```python
class Observer:
    def __init__(self, name):
        self.name = name
        self._messages = []

    def update(self, message):
        self._messages.append(f"{self.name} received: {message}")

    def received_messages(self):
        return list(self._messages)

class Observable:
    def __init__(self):
        self._observers = []

    def attach(self, observer):
        self._observers.append(observer)

    def detach(self, observer):
        self._observers.remove(observer)

    def notify(self, message):
        for observer in self._observers:
            observer.update(message)
```

**题目 2（dataclass 矩阵类）：**

```python
from dataclasses import dataclass, field

@dataclass
class Matrix:
    rows: int
    cols: int
    data: list = field(repr=False)

    @classmethod
    def from_2d(cls, grid):
        rows = len(grid)
        cols = len(grid[0]) if rows > 0 else 0
        data = [item for row in grid for item in row]
        return cls(rows, cols, data)

    def get(self, r, c):
        return self.data[r * self.cols + c]

    def set(self, r, c, value):
        self.data[r * self.cols + c] = value

    def to_2d(self):
        return [
            self.data[i * self.cols:(i + 1) * self.cols]
            for i in range(self.rows)
        ]

    def __add__(self, other):
        if self.rows != other.rows or self.cols != other.cols:
            raise ValueError("矩阵维度不匹配")
        data = [a + b for a, b in zip(self.data, other.data)]
        return Matrix(self.rows, self.cols, data)

    def __eq__(self, other):
        return (self.rows == other.rows and
                self.cols == other.cols and
                self.data == other.data)

    def __repr__(self):
        return f"Matrix({self.to_2d()})"
```

**题目 3（环境变量配置加载器）：**

```python
import os
from pathlib import Path

class EnvConfig:
    def __init__(self, env_file=None):
        if env_file and Path(env_file).exists():
            for line in Path(env_file).read_text(encoding="utf-8").splitlines():
                line = line.strip()
                if line and not line.startswith("#") and "=" in line:
                    key, value = line.split("=", 1)
                    os.environ[key.strip()] = value.strip()

    def get(self, key, default=None):
        return os.environ.get(key, default)

    def get_int(self, key, default=0):
        try:
            return int(os.environ.get(key, default))
        except (ValueError, TypeError):
            return default

    def get_bool(self, key, default=False):
        value = os.environ.get(key, "")
        if value.lower() in ("true", "1", "yes"):
            return True
        if value.lower() in ("false", "0", "no"):
            return False
        return default

    def require(self, key):
        value = os.environ.get(key)
        if value is None:
            raise EnvironmentError(f"缺少必需的环境变量: {key}")
        return value

    def as_dict(self, prefix="APP_"):
        return {
            k: v for k, v in os.environ.items()
            if k.startswith(prefix)
        }
```

**题目 4（异步任务管道）：**

```python
import asyncio

async def source(data, queue):
    for item in data:
        await queue.put(item)
    await queue.put(None)

async def transform(in_queue, out_queue, func):
    while True:
        item = await in_queue.get()
        if item is None:
            await out_queue.put(None)
            break
        await out_queue.put(func(item))

async def sink(in_queue, results):
    while True:
        item = await in_queue.get()
        if item is None:
            break
        results.append(item)

async def pipeline(raw_data, transforms, results):
    queues = [asyncio.Queue() for _ in range(len(transforms) + 2)]

    src_task = asyncio.create_task(source(raw_data, queues[0]))

    transform_tasks = []
    for i, func in enumerate(transforms):
        t = asyncio.create_task(transform(queues[i], queues[i + 1], func))
        transform_tasks.append(t)

    sink_task = asyncio.create_task(sink(queues[-1], results))

    await src_task
    await asyncio.gather(*transform_tasks)
    await sink_task
```

```python
import math
from datetime import datetime
from pydantic import BaseModel, field_validator, computed_field

class UserCreate(BaseModel):
    username: str
    email: str
    password: str
    age: int

    @field_validator("username")
    @classmethod
    def check_username(cls, v):
        if not 3 <= len(v) <= 20:
            raise ValueError("用户名长度必须在 3~20 之间")
        return v

    @field_validator("email")
    @classmethod
    def check_email(cls, v):
        if "@" not in v:
            raise ValueError("邮箱格式无效")
        return v

    @field_validator("password")
    @classmethod
    def check_password(cls, v):
        if len(v) < 8:
            raise ValueError("密码至少 8 个字符")
        return v

    @field_validator("age")
    @classmethod
    def check_age(cls, v):
        if not 0 <= v <= 150:
            raise ValueError("年龄必须在 0~150 之间")
        return v

class UserResponse(BaseModel):
    id: int
    username: str
    email: str
    created_at: str

    @classmethod
    def from_create(cls, user_id, user: UserCreate):
        return cls(
            id=user_id,
            username=user.username,
            email=user.email,
            created_at=datetime.now().isoformat(),
        )

class PaginatedResponse(BaseModel):
    items: list
    total: int
    page: int
    page_size: int

    @computed_field
    @property
    def total_pages(self) -> int:
        if self.page_size <= 0:
            return 0
        return math.ceil(self.total / self.page_size)
```
