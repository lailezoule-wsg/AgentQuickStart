# Python 装饰器学习指南

## 一、核心概念

装饰器本质上是一个**函数**，它接收一个函数作为参数，返回一个新的函数，从而在不修改原函数代码的前提下扩展其功能。

通俗理解：装饰器就像"包装纸"，给原函数包一层额外功能，原函数本身不变。

---

## 二、前置知识：函数是一等公民

在 Python 中，函数和普通变量一样，可以：
- 作为参数传递
- 作为返回值
- 赋值给变量

```python
def greet(name):
    return f"Hello, {name}"

say_hi = greet          # 赋值给变量
print(say_hi("Alice"))  # Hello, Alice

def call_func(func, name):   # 作为参数传递
    return func(name)

print(call_func(greet, "Bob"))  # Hello, Bob

def make_greeter(prefix):   # 作为返回值
    def inner(name):
        return f"{prefix}, {name}"
    return inner

hello = make_greeter("Hi")
print(hello("Carol"))   # Hi, Carol
```

这是理解装饰器的基础——装饰器就是利用"函数作为参数和返回值"实现的。

---

## 三、最简单的装饰器

```python
def my_decorator(func):
    def wrapper():
        print("函数执行前")
        func()
        print("函数执行后")
    return wrapper

@my_decorator
def say_hello():
    print("Hello!")

say_hello()
```

输出：
```
函数执行前
Hello!
函数执行后
```

`@my_decorator` 等价于 `say_hello = my_decorator(say_hello)`，这是一种语法糖。

---

## 四、装饰带参数和返回值的函数

实际中函数通常有参数和返回值，需要用 `*args, **kwargs` 适配任意签名。

```python
def logger(func):
    def wrapper(*args, **kwargs):
        print(f"调用 {func.__name__}，参数: {args}, {kwargs}")
        result = func(*args, **kwargs)
        print(f"{func.__name__} 执行完毕，返回: {result}")
        return result
    return wrapper

@logger
def add(a, b):
    return a + b

print(add(3, 5))
```

输出：
```
调用 add，参数: (3, 5), {}
add 执行完毕，返回: 8
8
```

## 五、functools.wraps 的作用

装饰后，原函数的元信息（名字、文档字符串）会丢失。用 `functools.wraps` 保留它们。

```python
from functools import wraps

def logger(func):
    @wraps(func)   # 加这一行
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@logger
def add(a, b):
    """两数相加"""
    return a + b

print(add.__name__)   # add（不加 wraps 会是 wrapper）
print(add.__doc__)    # 两数相加
```

**建议：写装饰器时永远加上 `@wraps(func)`。**

## 六、带参数的装饰器

如果装饰器本身需要参数，就要再嵌套一层：外层接收参数，中层接收函数，内层执行逻辑。

```python
from functools import wraps

def repeat(times):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            result = None
            for _ in range(times):
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

@repeat(3)
def greet(name):
    print(f"Hi, {name}")

greet("Alice")
```

输出：
```
Hi, Alice
Hi, Alice
Hi, Alice
```

执行流程：`repeat(3)` 返回 `decorator` → `decorator(greet)` 返回 `wrapper`。

## 七、类装饰器

用类实现装饰器，依赖 `__call__` 方法。

```python
from functools import wraps

class CountCalls:
    def __init__(self, func):
        wraps(func)(self)   # 保留元信息
        self.func = func
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"{self.func.__name__} 已被调用 {self.count} 次")
        return self.func(*args, **kwargs)

@CountCalls
def say_hi():
    print("Hi!")

say_hi()
say_hi()
```

输出：
```
say_hi 已被调用 1 次
Hi!
say_hi 已被调用 2 次
Hi!
```

类装饰器适合需要**维护状态**的场景（如计数、缓存）。

## 八、用装饰器装饰类

装饰器不仅能装饰函数，也能装饰类，用于添加方法或修改行为。

```python
def add_method(cls):
    cls.new_method = lambda self: f"我是 {self.__class__.__name__} 的新方法"
    return cls

@add_method
class MyClass:
    pass

obj = MyClass()
print(obj.new_method())   # 我是 MyClass 的新方法
```

## 九、常见内置装饰器

| 装饰器 | 作用 |
|--------|------|
| `@staticmethod` | 定义静态方法，无需实例 |
| `@classmethod` | 定义类方法，第一个参数是类本身 |
| `@property` | 把方法变成属性访问 |
| `@functools.lru_cache` | 自动缓存函数结果 |
| `@dataclass` | 自动生成 `__init__`、`__repr__` 等 |

```python
from functools import lru_cache

@lru_cache(maxsize=128)
def fib(n):
    if n < 2:
        return n
    return fib(n - 1) + fib(n - 2)

print(fib(100))   # 极快，结果被缓存
```

```python
class Circle:
    def __init__(self, radius):
        self._radius = radius

    @property
    def area(self):
        import math
        return math.pi * self._radius ** 2

c = Circle(5)
print(c.area)   # 像属性一样访问，无需括号
```

## 十、实战案例

### 案例 1：计时器

测量函数执行耗时。

```python
import time
from functools import wraps

def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} 耗时 {elapsed:.4f} 秒")
        return result
    return wrapper

@timer
def slow_sum(n):
    return sum(range(n))

print(slow_sum(1_000_000))
```

### 案例 2：日志记录

记录函数调用细节，便于调试。

```python
from functools import wraps
from datetime import datetime

def log(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        ts = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        print(f"[{ts}] 调用 {func.__name__}(args={args}, kwargs={kwargs})")
        try:
            result = func(*args, **kwargs)
            print(f"[{ts}] {func.__name__} 返回 {result}")
            return result
        except Exception as e:
            print(f"[{ts}] {func.__name__} 抛出异常: {e}")
            raise
    return wrapper

@log
def divide(a, b):
    return a / b

divide(10, 2)
divide(10, 0)
```

### 案例 3：权限校验

模拟用户权限检查，不通过则拒绝调用。

```python
from functools import wraps

current_user = {"name": "alice", "role": "guest"}

def require_role(role):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            if current_user.get("role") != role:
                raise PermissionError(f"需要 {role} 权限，当前为 {current_user.get('role')}")
            return func(*args, **kwargs)
        return wrapper
    return decorator

@require_role("admin")
def delete_user(user_id):
    print(f"已删除用户 {user_id}")

delete_user(1)   # guest 角色会抛 PermissionError
```

### 案例 4：自动重试

网络请求失败时自动重试若干次。

```python
import time
import random
from functools import wraps

def retry(times=3, delay=1):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exc = None
            for attempt in range(1, times + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exc = e
                    print(f"第 {attempt} 次失败: {e}")
                    if attempt < times:
                        time.sleep(delay)
            raise last_exc
        return wrapper
    return decorator

@retry(times=3, delay=0.5)
def fetch_data():
    if random.random() < 0.7:
        raise ConnectionError("网络波动")
    return "数据获取成功"

print(fetch_data())
```

### 案例 5：缓存结果

手写一个简易缓存装饰器（类似 `lru_cache` 的简化版）。

```python
from functools import wraps

def cache(func):
    stored = {}

    @wraps(func)
    def wrapper(*args):
        if args not in stored:
            stored[args] = func(*args)
        return stored[args]
    return wrapper

@cache
def square(n):
    print(f"计算 {n} 的平方")
    return n * n

print(square(4))   # 计算并缓存
print(square(4))   # 直接返回缓存
print(square(5))   # 计算并缓存
```

### 案例 6：限流

限制函数在一段时间内的调用次数。

```python
import time
from collections import deque
from functools import wraps

def rate_limit(calls, period):
    def decorator(func):
        history = deque()

        @wraps(func)
        def wrapper(*args, **kwargs):
            now = time.time()
            while history and history[0] < now - period:
                history.popleft()
            if len(history) >= calls:
                raise RuntimeError(f"超过限流：{calls} 次 / {period} 秒")
            history.append(now)
            return func(*args, **kwargs)
        return wrapper
    return decorator

@rate_limit(calls=3, period=10)
def send_msg(msg):
    print(f"发送: {msg}")

for i in range(5):
    try:
        send_msg(f"消息{i}")
    except RuntimeError as e:
        print(e)
```

## 十一、装饰器叠加

多个装饰器可叠加使用，**执行顺序从下到上装饰，从上到下执行**。

```python
from functools import wraps

def dec_a(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print("A 前")
        result = func(*args, **kwargs)
        print("A 后")
        return result
    return wrapper

def dec_b(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print("B 前")
        result = func(*args, **kwargs)
        print("B 后")
        return result
    return wrapper

@dec_a
@dec_b
def hello():
    print("hello")

hello()
```

输出：
```
A 前
B 前
hello
B 后
A 后
```

相当于 `hello = dec_a(dec_b(hello))`。

## 十二、要点总结

1. 装饰器 = 接收函数、返回函数的函数，用 `@` 语法应用。
2. 用 `*args, **kwargs` 让 wrapper 适配任意函数签名。
3. **永远加 `@wraps(func)`** 保留原函数元信息。
4. 带参数的装饰器需要三层嵌套：参数层 → 函数层 → wrapper 层。
5. 类装饰器靠 `__call__`，适合维护状态。
6. 装饰器可叠加，顺序：装饰从下到上，执行从外到内。
7. 常用场景：日志、计时、缓存、权限、重试、限流。

建议按顺序敲一遍以上代码，理解后再尝试自己写一个装饰器。
