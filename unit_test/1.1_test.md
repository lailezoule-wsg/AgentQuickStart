def format_table(rows, headers):
    col_widths = []
    for i, h in enumerate(headers):
        max_w = len(str(h))

        col_widths.append(max_w)

    for row in rows:

## 简答题答案

**1.** `list` 用于需要动态修改的有序集合，如 `items = []; items.append("new")`。`tuple` 用于不应被修改的固定数据，如坐标 `point = (10, 20)` 或函数返回多个值。

**2.** `==` 比较值是否相等，`is` 比较是否是同一个对象（内存地址）。
```python
a = [1, 2, 3]
b = [1, 2, 3]
a == b  # True
a is b  # False
```

**3.** 因为默认参数在函数定义时只创建一次，可变对象会在多次调用间共享状态：
```python
def add_item(item, lst=[]):
    lst.append(item)
    return lst
print(add_item(1))  # [1]
print(add_item(2))  # [1, 2] —— 不是期望的 [2]！
```
应改为 `lst=None`，函数内 `if lst is None: lst = []`。

**4.** Python 推荐 EAFP（Easier to Ask Forgiveness than Permission），即优先用 `try/except`。因为：①代码更简洁；②多线程环境下更安全；③正常路径性能更好。

def format_table(rows, headers):
    col_widths = []
    for i, h in enumerate(headers):
        max_w = len(str(h))
        for row in rows:
            max_w = max(max_w, len(str(row[i])))
        col_widths.append(max_w)

    def fmt_row(values):
        return " | ".join(str(v).ljust(col_widths[i]) for i, v in enumerate(values))

    lines = [fmt_row(headers)]

    for row in rows:

    items = {}
    for k, v in d.items():
        new_key = f"{parent_key}{sep}{k}" if parent_key else k

import time

def retry(func, max_attempts=3, delay=0):
    last_exception = None
    for attempt in range(1, max_attempts + 1):
        try:
            return func()
        except Exception as e:
            last_exception = e
            if attempt < max_attempts:
